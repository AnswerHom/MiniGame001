#### 4.1.2 引导UI实现
```javascript
function showTutorialStep(stepIndex) {
  const step = tutorial.steps[stepIndex];

  // 创建引导弹窗
  const dialog = document.createElement('div');
  dialog.className = 'tutorial-dialog';
  dialog.innerHTML = `
    <div class="tutorial-content">
      <h3 class="tutorial-title">${step.title}</h3>
      <p class="tutorial-description">${step.description}</p>
      <div class="tutorial-buttons">
        <button class="btn btn-skip" onclick="skipTutorial()">跳过</button>
        <button class="btn btn-next" onclick="nextTutorialStep()">下一步</button>
      </div>
    </div>
  `;

  // 高亮目标元素
  if (step.target) {
    const targetElement = document.getElementById(step.target);
    if (targetElement) {
      targetElement.classList.add('tutorial-highlight');
    }
  }

  document.body.appendChild(dialog);
}

function nextTutorialStep() {
  // 隐藏当前步骤
  const currentDialog = document.querySelector('.tutorial-dialog');
  if (currentDialog) {
    currentDialog.remove();
  }

  // 隐藏高亮
  document.querySelectorAll('.tutorial-highlight').forEach(el => {
    el.classList.remove('tutorial-highlight');
  });

  tutorial.currentStep++;

  // 检查是否完成
  if (tutorial.currentStep >= tutorial.steps.length) {
    tutorial.completed = true;
    showMessage('🎉 恭喜！你已经完成了新手引导！');
  } else {
    showTutorialStep(tutorial.currentStep);
  }
}

function skipTutorial() {
  tutorial.completed = true;
  const currentDialog = document.querySelector('.tutorial-dialog');
  if (currentDialog) {
    currentDialog.remove();
  }
  document.querySelectorAll('.tutorial-highlight').forEach(el => {
    el.classList.remove('tutorial-highlight');
  });
}
```

### 4.3 顾客购买逻辑优化

#### 4.3.1 智能购买决策
```javascript
function customerDecidePurchase(customer) {
  // 计算每件商品的吸引力分数
  const products = gameState.products.filter(p => p.stock > 0);

  if (products.length === 0) {
    return null;
  }

  const productScores = products.map(product => {
    const attractivenessScore = calculateAttractivenessScore(product, customer);
    return { product, score: attractivenessScore };
  });

  // 选择分数最高的商品
  productScores.sort((a, b) => b.score - a.score);
  return productScores[0].product;
}

function calculateAttractivenessScore(product, customer) {
  // 库存充足度：库存越多，分数越高
  const stockScore = Math.min(1, product.stock / 20);

  // 价格合适度：价格越接近顾客期望，分数越高
  const priceScore = calculatePriceScore(product, customer);

  // 综合吸引力分数
  return stockScore * 0.4 + priceScore * 0.6;
}

function calculatePriceScore(product, customer) {
  const expectedPrice = product.basePrice * (1 + (Math.random() - 0.5) * 0.5);

  if (expectedPrice >= customer.minPrice && expectedPrice <= customer.maxPrice) {
    return 1.0;
  } else if (expectedPrice < customer.minPrice) {
    return 0.5;
  } else {
    return 0.3;
  }
}
```

## 七、风险点

1. **引导过于冗长**：可能让玩家失去耐心
   - 解决方案：可跳过，每个步骤简洁明了

2. **目标难度不平衡**：某些目标可能太难或太简单
   - 解决方案：持续测试和调整

3. **升级建议可能过于频繁**：可能影响游戏体验
   - 解决方案：合理设置建议频率

4. **智能购买逻辑过于复杂**：可能影响性能
   - 解决方案：优化算法，减少不必要的计算
