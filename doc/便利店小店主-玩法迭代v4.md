- 库存充足（>0）：满意度 100%
- 库存不足：满意度 60%

服务速度：
- 收银台等级越高，服务速度越快
- VIP顾客对服务速度要求更高

## 五、新增：事件系统（v4核心）

### 5.1 事件系统概述

事件系统是游戏的核心机制，通过随机触发各类事件，增加游戏的不确定性和策略深度。

**事件系统设计原则：**
- 多样性：多种事件类型，丰富游戏体验
- 可控性：玩家可以通过策略影响事件触发概率
- 奖惩机制：事件带来金币奖励或损失
- 难度递进：事件难度随游戏等级提升

### 5.2 事件分类

#### 5.2.1 常规事件（日常型）

**事件1：顾客投诉**
- **触发条件：**
  - 排队人数 > 4
  - 顾客等待时间 > 5秒
  - 收银台等级 < 2
- **事件描述：** "一位顾客在排队时因为等待时间过长而生气离开了"
- **影响：**
  - 满意度 -30
  - 失去一个顾客
  - 潜在流失率 +10%
- **应对策略：**
  - 升级收银台
  - 优化排队管理
  - 增加收银员

**事件2：商品缺货**
- **触发条件：**
  - 某类商品库存为 0

3. 特殊事件（中优先级）
4. 常规事件（低优先级）

#### 5.3.3 事件冷却时间
- 每个事件类型有独立的冷却时间
- 冷却时间随游戏等级提升而缩短
- 冷却时间结束后才能再次触发

### 5.5 事件系统实现

#### 5.5.1 数据结构
```javascript
// 事件系统数据结构
class EventSystem {
  constructor() {
    this.events = {}; // 所有事件定义
    this.activeEvents = []; // 当前激活的事件
    this.eventCooldowns = {}; // 事件冷却时间
    this.eventHistory = []; // 事件历史记录
  }

  // 注册事件
  registerEvent(event) {
    this.events[event.id] = event;
  }

  // 触发事件
  triggerEvent(eventId) {
    const event = this.events[eventId];

    // 检查冷却时间
    if (this.eventCooldowns[eventId] > 0) {
      return false;
    }

    // 检查触发条件
    if (!event.triggerCondition()) {
      return false;
    }

    // 创建事件实例
    const eventInstance = {
      ...event,
      id: eventId,
      startTime: Date.now(),
      duration: event.duration,
      status: 'active'
    };

    this.activeEvents.push(eventInstance);
    this.eventCooldowns[eventId] = event.cooldown;

    // 通知玩家
    this.notifyPlayer(eventInstance);

    return true;
  }

  // 更新事件
  update() {
    const now = Date.now();

    // 更新冷却时间
    for (const eventId in this.eventCooldowns) {
      this.eventCooldowns[eventId] -= 1000; // 1秒
    }

    // 检查事件是否结束
    this.activeEvents = this.activeEvents.filter(event => {
      if (now - event.startTime > event.duration) {
        this.endEvent(event);
        return false;
      }
      return true;
    });
  }

  // 结束事件
  endEvent(event) {
    event.status = 'ended';
    this.eventHistory.push(event);
  }

  // 通知玩家
  notifyPlayer(event) {
    // 显示事件提示
    showEventNotification(event);
  }
}
```

// 事件触发器
class EventTrigger {
  constructor(eventSystem) {
    this.eventSystem = eventSystem;
  }

  // 定时触发器
  startTimer() {
    setInterval(() => {
      this.checkEventTriggers();
    }, 3000); // 每3秒检查一次
  }

  // 检查事件触发条件
  checkEventTriggers() {
    // 检查常规事件
    if (Math.random() < 0.2) { // 20% 概率
      this.eventSystem.triggerEvent('customer_complaint');
    }

    // 检查特殊事件

      this.eventSystem.triggerEvent('promotional_activity');

    // 检查挑战事件
    if (Math.random() < 0.05) { // 5% 概率
      this.eventSystem.triggerEvent('inventory_crisis');
    }
  }
}

```javascript
// eventSystem.js
import { events } from './events.js';

class EventSystem {
  constructor() {
    this.eventDefinitions = events;
    this.activeEvents = [];
    this.eventCooldowns = {};
    this.eventHistory = [];
    this.notificationTimer = null;
  }

  // 初始化事件系统
  initialize() {
    this.startNotificationTimer();
  }

  // 启动事件触发器
  startEventTrigger() {
    setInterval(() => {
      this.checkAndTriggerEvents();
    }, 3000); // 每3秒检查一次
  }

  // 检查并触发事件
  checkAndTriggerEvents() {
    for (const eventId in this.eventDefinitions) {
      const event = this.eventDefinitions[eventId];

      // 检查冷却时间
      if (this.eventCooldowns[eventId] > 0) {
        this.eventCooldowns[eventId] -= 3000;
        continue;
      }

      // 检查触发条件
      if (event.triggerCondition()) {
        this.triggerEvent(eventId);
      }
    }
  }

  // 触发事件
  triggerEvent(eventId) {
    const event = this.eventDefinitions[eventId];

    // 创建事件实例
    const eventInstance = {
      ...event,
      id: eventId,
      startTime: Date.now(),
      endTime: Date.now() + event.duration,
      status: 'active'
    };

    this.activeEvents.push(eventInstance);
    this.eventCooldowns[eventId] = event.cooldown;

    // 通知玩家
    this.showEventNotification(eventInstance);

    // 应用事件效果
    this.applyEventEffects(event);
  }

  // 显示事件通知
  showEventNotification(event) {
    const notification = {
      type: event.type,
      title: event.name,
      description: event.description,
      effects: event.effects,
      strategies: event.strategies,
      duration: event.duration,
      startTime: Date.now()
    };

    // 显示通知栏
    showEventNotificationBar(notification);

    // 显示对话框
    showEventDialog(notification);
  }

  // 应用事件效果
  applyEventEffects(event) {
    // 应用效果到游戏系统
    if (event.effects.revenue) {
      game.revenue *= event.effects.revenue;
    }

    if (event.effects.satisfaction !== undefined) {
      game.satisfaction += event.effects.satisfaction;
    }

    if (event.effects.goldReward) {
      game.gold += event.effects.goldReward;
    }

    if (event.effects.goldLoss) {
      game.gold = Math.max(0, game.gold - event.effects.goldLoss);
    }
  }

  // 更新事件状态
  updateEvents() {
    const now = Date.now();

    this.activeEvents = this.activeEvents.filter(event => {
      if (now >= event.endTime) {
        this.endEvent(event);
        return false;
      }
      return true;
    });
  }

  // 结束事件
  endEvent(event) {
    event.status = 'ended';
    this.eventHistory.push(event);

    // 显示事件结束通知
    showEventEndNotification(event);
  }

  // 应用应对策略
  applyStrategy(eventId, strategyIndex) {
    const event = this.eventDefinitions[eventId];
    const strategy = event.strategies[strategyIndex];

    if (strategy.action === 'upgrade_cashier') {
      game.upgradeCashierStation(strategy.cost);
    } else if (strategy.action === 'add_cashier') {
      game.addCashier(strategy.cost);
    } else if (strategy.action === 'emergency_restock') {
      game.emergencyRestock(strategy.cost);
    }
  }

  // 获取事件统计
  getEventStats() {
    return {
      totalEvents: Object.keys(this.eventDefinitions).length,
      activeEvents: this.activeEvents.length,
      eventHistory: this.eventHistory.length,
      eventTypes: {
        regular: 0,
        special: 0,
        challenge: 0
      }
    };
  }
}

export default EventSystem;
```

class QueueManager {
  constructor() {
    this.queue = []; // 排队顾客列表
    this.maxSize = 8; // 最大排队人数
  }

  // 添加顾客到排队
  addCustomer(customer) {
    if (this.queue.length >= this.maxSize) {
      return false; // 排队已满
    }

    // VIP顾客插队
    if (customer.priority === 3) {
      this.queue.unshift(customer);
    } else {
      this.queue.push(customer);
    }
    return true;
  }

  // 获取下一个服务的顾客
  getNextCustomer() {
    return this.queue.shift(); // 优先级高的在前
  }

  // 顾客离开
  removeCustomer(customerId) {
    this.queue = this.queue.filter(c => c.id !== customerId);
  }
}
